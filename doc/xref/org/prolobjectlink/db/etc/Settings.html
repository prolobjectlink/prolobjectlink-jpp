149 	public Settings load() {
150 		try {
151 			File configuration = getConfiguration();
152 			properties.loadFromXML(new FileInputStream(configuration));
153 			String driver = properties.getProperty(FACTORY);
154 			String provider = properties.getProperty(PROVIDER);
155 			Class<?> clazzDriver = JavaReflect.classForName(driver);
156 			Class<?> clazzProvider = JavaReflect.classForName(provider);
157 
158 			this.timeGranularity = Integer
159 					.valueOf((String) properties.getProperty(TIME_GRANULARITY, "" + DEFAULT_TIME_GRANULARITY + ""));
160 			this.lockSleep = Integer.valueOf((String) properties.getProperty(LOCK_SLEEP, "" + DEFAULT_LOCK_SLEEP + ""));
161 			this.sleepGap = Integer.valueOf((String) properties.getProperty(SLEEP_GAP, "" + DEFAULT_SLEEP_GAP + ""));
162 			this.port = Integer.valueOf((String) properties.getProperty(PORT, "" + DEFAULT_PORT + ""));
163 			this.password = properties.getProperty(SECRET, DEFAULT_SECRET);
164 			this.username = properties.getProperty(USER, DEFAULT_USER);
165 
166 			String storage = properties.getProperty(STORAGE);
167 			if (storage.equals("" + StorageMode.SINGLE_STORAGE + "")) {
168 				this.storageMode = StorageMode.SINGLE_STORAGE;
169 			} else {
170 				this.storageMode = StorageMode.STORAGE_POOL;
171 			}
172 			prologProvider = (PrologProvider) JavaReflect.newInstance(clazzProvider);
173 			containerFactory = (ContainerFactory) JavaReflect.newInstance(clazzDriver);
174 			containerFactory.setProvider(prologProvider);
175 			containerFactory.setSettings(this);
176 		} catch (FileNotFoundException e) {
177 			LoggerUtils.error(getClass(), LoggerConstants.FILE_NOT_FOUND, e);
178 		} catch (IOException e) {
179 			LoggerUtils.error(getClass(), LoggerConstants.IO, e);
180 		}
181 		return this;
182 	}

184 	public Settings save() {
185 		try {
186 			properties.put(PROVIDER, prologProvider.getClass().getName());
187 			properties.put(FACTORY, containerFactory.getClass().getName());
188 			properties.put(TIME_GRANULARITY, "" + timeGranularity + "");
189 			properties.put(LOCK_SLEEP, "" + lockSleep + "");
190 			properties.put(SLEEP_GAP, "" + sleepGap + "");
191 			properties.put(STORAGE, "" + storageMode + "");
192 			properties.put(PORT, "" + port + "");
193 			properties.put(SECRET, password);
194 			properties.put(USER, username);
195 			File configuration = getConfiguration();
196 			properties.storeToXML(new FileOutputStream(configuration), null);
197 		} catch (IOException e) {
198 			LoggerUtils.error(getClass(), LoggerConstants.IO, e);
199 		}
200 		return this;
201 	}

392 	public final File getConfiguration() throws IOException {
393 		File configuration = null;
394 		String folder = getCurrentPath();
395 		File plk = new File(folder);
396 		File pdk = plk.getParentFile();
397 		File prt = pdk.getParentFile();
398 		if (prt.getCanonicalPath().contains("prolobjectlink-jpp")) { // dev mode
399 			configuration = new File(CONFIG);
400 		} else { // production mode
401 			configuration = new File(prt.getCanonicalPath() + File.separator + CONFIG);
402 		}
403 		return configuration;
404 	}
