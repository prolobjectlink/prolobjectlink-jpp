61  public final class RemoteDatabaseThread extends AbstractWrapper implements Runnable, Wrapper {
62  
63  	private final int id;
64  	private Socket socket;
65  	private DatabaseEngine database;
66  	private RemoteDatabaseServer server;

68  	private static final Restore restorer = new Restore();
69  	private final EnumMap<DatabaseType, Map<String, DatabaseEngine>> dbTypes;

71  	RemoteDatabaseThread(int id, Socket socket, RemoteDatabaseServer server) {
72  		this.dbTypes = new EnumMap<DatabaseType, Map<String, DatabaseEngine>>(DatabaseType.class);
73  		this.socket = socket;
74  		this.server = server;
75  		this.id = id;
76  	}

102 	public void run() {
103 
104 		try {
105 
106 			LoggerUtils.debug(getClass(), "request " + id + " at " + socket.getInetAddress().getHostAddress()
107 					+ " on port " + socket.getPort());
108 
109 			DatabaseResponse response = new DatabaseResponse();
110 
111 			ObjectOutputStream os = new ObjectOutputStream(socket.getOutputStream());
112 			ObjectInputStream is = new ObjectInputStream(socket.getInputStream());
113 
114 			// read data from client
115 			DatabaseRequest request = unwrap(is.readObject(), DatabaseRequest.class);
116 
117 			LoggerUtils.debug(getClass(), request);
118 
119 			// server database operations area
120 			// (insert,update,delete,query,...)
121 
122 			String dir = null;
123 			String zip = null;
124 			String string = null;
125 			boolean exist = false;
126 			Class<?> clazz = null;
127 			Transaction tx = null;
128 			Predicate<?> p = null;
129 			Serializable[] sa = null;
130 			PersistentContainer pc = null;
131 
132 			switch (request.getType()) {
133 			case CONNECT: // OK
134 				DatabaseType t = request.getDatabaseType();
135 				Map<String, DatabaseEngine> dbs = dbTypes.get(t);
136 				database = dbs.get(request.getDatabaseName());
137 				if (database == null) {
138 					response.set(false);
139 				} else {
140 					response.set(true);
141 				}
142 				os.writeObject(response);
143 				LoggerUtils.debug(getClass(), response);
144 				break;
145 			case CREATE:
146 				database.create();
147 				response.setVoid();
148 				os.writeObject(response);
149 				LoggerUtils.debug(getClass(), response);
150 				break;
151 			case CLEAR:
152 				tx = database.getTransaction();
153 				if (!tx.isActive()) {
154 					tx.begin();
155 				}
156 				database.clear();
157 				tx.commit();
158 				tx.close();
159 				response.setVoid();
160 				os.writeObject(response);
161 				LoggerUtils.debug(getClass(), response);
162 				break;
163 			case DROP:
164 				database.drop();
165 				response.setVoid();
166 				os.writeObject(response);
167 				LoggerUtils.debug(getClass(), response);
168 				break;
169 			case BEGIN: // OK
170 				tx = database.getTransaction();
171 				if (!tx.isActive()) {
172 					tx.begin();
173 				}
174 				response.setVoid();
175 				os.writeObject(response);
176 				LoggerUtils.debug(getClass(), response);
177 				break;
178 			case INSERT: // OK
179 				clazz = (Class<?>) request.getArgument(0);
180 				sa = (Serializable[]) request.getArgument(1);
181 				pc = database.containerOf(clazz);
182 				tx = pc.getTransaction();
183 				if (!tx.isActive()) {
184 					tx.begin();
185 				}
186 				pc.insert(sa);
187 				tx.commit();
188 				tx.close();
189 				response.setVoid();
190 				os.writeObject(response);
191 				LoggerUtils.debug(getClass(), response);
192 				break;
193 			case UPDATE: // OK
194 				clazz = (Class<?>) request.getArgument(0);
195 				sa = (Serializable[]) request.getArgument(1);
196 				if (sa.length == 2) {
197 					pc = database.containerOf(clazz);
198 					tx = pc.getTransaction();
199 					if (!tx.isActive()) {
200 						tx.begin();
201 					}
202 					pc.update(sa[0], sa[1]);
203 					tx.commit();
204 					tx.close();
205 				}
206 				response.setVoid();
207 				os.writeObject(response);
208 				LoggerUtils.debug(getClass(), response);
209 				break;
210 			case DELETE_ARRAY: // OK
211 				clazz = (Class<?>) request.getArgument(0);
212 				sa = (Serializable[]) request.getArgument(1);
213 				pc = database.containerOf(clazz);
214 				tx = pc.getTransaction();
215 				if (!tx.isActive()) {
216 					tx.begin();
217 				}
218 				pc.delete(sa);
219 				tx.commit();
220 				tx.close();
221 				response.setVoid();
222 				os.writeObject(response);
223 				LoggerUtils.debug(getClass(), response);
224 				break;
225 			case DELETE_CLASS:// OK
226 				clazz = (Class<?>) request.getArgument(0);
227 				pc = database.containerOf(clazz);
228 				tx = pc.getTransaction();
229 				if (!tx.isActive()) {
230 					tx.begin();
231 				}
232 				pc.delete(clazz);
233 				tx.commit();
234 				tx.close();
235 				response.setVoid();
236 				os.writeObject(response);
237 				LoggerUtils.debug(getClass(), response);
238 				break;
239 			case QUERY_STRING: // OK
240 				string = (String) request.getArgument(0);
241 				pc = database.getContainer();
242 				tx = pc.getTransaction();
243 				if (!tx.isActive()) {
244 					tx.begin();
245 				}
246 				Query k = pc.createQuery(string);
247 				tx.commit();
248 				tx.close();
249 				response.set(k);
250 				os.writeObject(response);
251 				LoggerUtils.debug(getClass(), response);
252 				break;
253 			case QUERY_OBJECT: // OK
254 				Object s = request.getArgument(0);
255 				pc = database.getContainer();
256 				tx = pc.getTransaction();
257 				if (!tx.isActive()) {
258 					tx.begin();
259 				}
260 				TypedQuery<?> u = pc.createQuery(s);
261 				tx.commit();
262 				tx.close();
263 				response.set(u);
264 				os.writeObject(response);
265 				LoggerUtils.debug(getClass(), response);
266 				break;
267 			case QUERY_CLASS: // OK
268 				clazz = (Class<?>) request.getArgument(0);
269 				pc = database.containerOf(clazz);
270 				tx = pc.getTransaction();
271 				if (!tx.isActive()) {
272 					tx.begin();
273 				}
274 				TypedQuery<?> q = pc.createQuery(clazz);
275 				tx.commit();
276 				tx.close();
277 				response.set(q);
278 				os.writeObject(response);
279 				LoggerUtils.debug(getClass(), response);
280 				break;
281 			case QUERY_PREDICATE: // OK
282 				clazz = (Class<?>) request.getArgument(0);
283 				p = (Predicate<?>) request.getArgument(1);
284 				pc = database.containerOf(clazz);
285 				tx = pc.getTransaction();
286 				if (!tx.isActive()) {
287 					tx.begin();
288 				}
289 				q = pc.createQuery(p);
290 				tx.commit();
291 				tx.close();
292 				response.set(q);
293 				os.writeObject(response);
294 				LoggerUtils.debug(getClass(), response);
295 				break;
296 			case PROCEDURE:
297 				Serializable a1 = request.getArgument(0);
298 				Serializable a2 = request.getArgument(1);
299 				String functor = (String) a1;
300 				String[] args = (String[]) a2;
301 				response.set(database.createProcedureQuery(functor, args));
302 				os.writeObject(response);
303 				LoggerUtils.debug(getClass(), response);
304 				break;
305 			case CONSTAINS_STRING: // OK
306 				string = (String) request.getArgument(0);
307 				pc = database.getContainer();
308 				tx = pc.getTransaction();
309 				if (!tx.isActive()) {
310 					tx.begin();
311 				}
312 				exist = pc.contains(string);
313 				tx.commit();
314 				tx.close();
315 				response.set(exist);
316 				os.writeObject(response);
317 				LoggerUtils.debug(getClass(), response);
318 				break;
319 			case CONSTAINS_OBJECT: // OK
320 				Object h = request.getArgument(0);
321 				pc = database.getContainer();
322 				tx = pc.getTransaction();
323 				if (!tx.isActive()) {
324 					tx.begin();
325 				}
326 				exist = pc.contains(h);
327 				tx.commit();
328 				tx.close();
329 				response.set(exist);
330 				os.writeObject(response);
331 				LoggerUtils.debug(getClass(), response);
332 				break;
333 			case CONSTAINS_CLASS: // OK
334 				clazz = (Class<?>) request.getArgument(0);
335 				pc = database.containerOf(clazz);
336 				tx = pc.getTransaction();
337 				if (!tx.isActive()) {
338 					tx.begin();
339 				}
340 				exist = pc.contains(clazz);
341 				tx.commit();
342 				tx.close();
343 				response.set(exist);
344 				os.writeObject(response);
345 				LoggerUtils.debug(getClass(), response);
346 				break;
347 			case CONSTAINS_PREDICATE: // OK
348 				clazz = (Class<?>) request.getArgument(0);
349 				p = (Predicate<?>) request.getArgument(1);
350 				pc = database.containerOf(clazz);
351 				tx = pc.getTransaction();
352 				if (!tx.isActive()) {
353 					tx.begin();
354 				}
355 				exist = pc.contains(p);
356 				tx.commit();
357 				tx.close();
358 				response.set(exist);
359 				os.writeObject(response);
360 				LoggerUtils.debug(getClass(), response);
361 				break;
362 			case CONSTAINS_INDICATOR: // OK
363 				string = (String) request.getArgument(0);
364 				int arity = (Integer) request.getArgument(1);
365 				pc = database.getContainer();
366 				tx = pc.getTransaction();
367 				if (!tx.isActive()) {
368 					tx.begin();
369 				}
370 				exist = pc.contains(string, arity);
371 				tx.commit();
372 				tx.close();
373 				response.set(exist);
374 				os.writeObject(response);
375 				LoggerUtils.debug(getClass(), response);
376 				break;
377 			case INCLUDE:
378 				String path = (String) request.getArgument(0);
379 				StringBuilder pl = (StringBuilder) request.getArgument(1);
380 				// create file at some location in server
381 				String location = "upl" + path;
382 				File toBeInclude = new File(location);
383 				PrintWriter w = new PrintWriter(toBeInclude);
384 				w.write("" + pl + "");
385 				w.close();
386 				database.getContainer().include(location);
387 				response.setVoid();
388 				os.writeObject(response);
389 				LoggerUtils.debug(getClass(), response);
390 				break;
391 			case BACKUP: // OK
392 				dir = (String) request.getArgument(0);
393 				zip = (String) request.getArgument(1);
394 				String dbname = request.getDatabaseName();
395 				String a = database.getStorageLocation();
396 				String b = a.substring(0, a.indexOf(dbname));
397 				backuper.createBackup(b, dir, zip);
398 				response.setVoid();
399 				os.writeObject(response);
400 				LoggerUtils.debug(getClass(), response);
401 				break;
402 			case RESTORE: // OK
403 				dir = (String) request.getArgument(0);
404 				zip = (String) request.getArgument(1);
405 				restorer.restoreBackup(dir, zip);
406 				response.setVoid();
407 				os.writeObject(response);
408 				LoggerUtils.debug(getClass(), response);
409 				break;
410 			case COMMIT: // OK don't do any thing
411 				response.setVoid();
412 				os.writeObject(response);
413 				LoggerUtils.debug(getClass(), response);
414 				break;
415 			case ROLLBACK: // ???
416 				// TODO i need save every db change
417 				// and over roll-back request
418 				// load every db change and restore
419 				// before state. I need a change log.
420 				database.rollback();
421 				response.setVoid();
422 				os.writeObject(response);
423 				LoggerUtils.debug(getClass(), response);
424 				break;
425 			case CLOSE: // ??? No effect occurs
426 				database.close();
427 				response.setVoid();
428 				os.writeObject(response);
429 				LoggerUtils.debug(getClass(), response);
430 				break;
431 			default:
432 
433 				break;
434 			}
435 
436 			// close in/out streams
437 			is.close();
438 			os.close();
439 
440 		} catch (IOException e) {
441 			LoggerUtils.error(getClass(), LoggerConstants.IO, e);
442 		} catch (ClassNotFoundException e) {
443 			LoggerUtils.error(getClass(), LoggerConstants.CLASS_NOT_FOUND, e);
444 		}
