125 	public final Class<?> classOf(String functor, int arity) {
126 		String className = removeQuoted(functor);
127 		Class<?> clazz = JavaReflect.classForName(className);
128 		if (clazz.getDeclaredFields().length != arity) {
129 			try {
130 				throw new IllegalArgumentException("Non exist " + className + " with " + arity + " fields");
131 			} catch (IllegalArgumentException e) {
132 				LoggerUtils.error(getClass(), LoggerConstants.ILLEGAL_ARGUMENT, e);
133 			}
134 		}
135 		return clazz;
136 	}

138 	public final <O> Class<O> classOf(Predicate<O> predicate) {
139 		Class<O> templateClass = null;
140 		Class<?> clazz = predicate.getClass();
141 		Type[] generics = clazz.getGenericInterfaces();
142 		if (generics.length == 1 && generics[0] instanceof ParameterizedType) {
143 			ParameterizedType parameterized = (ParameterizedType) generics[0];
144 			Type type = parameterized.getActualTypeArguments()[0];
145 			if (!(type instanceof Class<?>)) {
146 				try {
147 					throw new ClassNotFoundException("" + type + "");
148 				} catch (ClassNotFoundException e) {
149 					LoggerUtils.error(getClass(), LoggerConstants.CLASS_NOT_FOUND, e);
150 				}
151 			}
152 			templateClass = (Class<O>) type;
153 		}
154 		return templateClass;
155 	}

165 	public final Object[] solutionOf(PrologTerm[] prologTerms, List<Class<?>> classes) {
166 		PrologQuery query = engine.query(prologTerms);
167 		if (query.hasSolution()) {
168 			Map<String, PrologTerm> solutionsMap = query.oneVariablesSolution();
169 			Object[] objects = new Object[classes.size()];
170 			for (int i = 0; i < classes.size(); i++) {
171 				objects[i] = converter.toObject(classes.get(i), solutionsMap);
172 			}
173 			return objects;
174 		}
175 		return new Object[0];
176 	}

178 	/**
179 	 * Find all objects solutions of the prolog terms array. Use a given list of
180 	 * classes extracted from prolog terms array for convert the result obtained
181 	 * from prolog query result.
182 	 * 
183 	 * @param prologTerms prolog terms array to query.
184 	 * 
185 	 * @param classes     classes extracted from prolog terms array.
186 	 * @return all objects solutions of prologTerms prolog terms array.
187 	 */
188 	public final List<Object> solutionsOf(PrologTerm[] prologTerms, List<Class<?>> classes) {
189 		List<Object> solutionSet = new ArrayList<Object>();
190 		PrologQuery query = engine.query(prologTerms);
191 		if (query.hasSolution()) {
192 			Map<String, PrologTerm>[] solutionsMap = query.allVariablesSolutions();
193 			for (int i = 0; i < solutionsMap.length; i++) {
194 				Object[] objects = new Object[classes.size()];
195 				for (int j = 0; j < classes.size(); j++) {
196 					objects[j] = converter.toObject(classes.get(j), solutionsMap[i]);
197 				}
198 				solutionSet.add(objects);
199 			}
200 		}
201 		return solutionSet;
202 	}

212 	protected final List<Class<?>> classesOf(PrologTerm[] prologTerms, List<Class<?>> classList) {
213 		for (int i = 0; i < prologTerms.length; i++) {
214 			PrologTerm prologTerm = prologTerms[i];
215 			if (prologTerm.isStructure()) {
216 				String functor = prologTerms[i].getFunctor();
217 				PrologTerm[] args = prologTerm.getArguments();
218 				if (!prologTerm.isEvaluable()) {
219 					String className = removeQuoted(functor);
220 					Class<?> clazz = JavaReflect.classForName(className);
221 					if (!classList.contains(clazz)) {
222 						classList.add(clazz);
223 					}
224 				}
225 				classesOf(args, classList);
226 			}
227 		}
228 		return classList;
229 	}

231 	protected final String removeQuoted(String functor) {
232 		if (functor != null && functor.startsWith("\'") && functor.endsWith("\'")) {
233 			return functor.substring(1, functor.length() - 1);
234 		}
235 		return functor;
236 	}

251 	protected final void checkProcedureInvokation(String name, Field[] fields, Object[] values) {
252 		int fieldsNumber = fields.length;
253 		int valuesNumber = values.length;
254 		if (fieldsNumber != valuesNumber) {
255 			throw new ProcedureInvokationError(name, fieldsNumber, valuesNumber);
256 		}
257 	}
