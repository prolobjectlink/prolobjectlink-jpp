33  package org.prolobjectlink.db.prolog;

35  import java.lang.reflect.Field;
36  
37  import org.prolobjectlink.db.Renamer;
38  import org.prolobjectlink.db.util.Assertions;
39  import org.prolobjectlink.db.util.JavaReflect;
40  import org.prolobjectlink.db.util.JavaStrings;
41  import org.prolobjectlink.prolog.PrologProvider;
42  import org.prolobjectlink.prolog.PrologVariable;
43  
44  final class PrologRenamer extends AbstractRenamer implements Renamer {
45  
46  	PrologRenamer(PrologProvider provider) {
47  		super(provider);
48  	}
49  
50  	public final PrologVariable toVariable(Field field) {
51  		Field workField = Assertions.requireNotNull(field);
52  		Class<?> workClass = workField.getDeclaringClass();
53  		String className = workClass.getName();
54  		String name = className.replace('.', '_');
55  		name = JavaStrings.toUpperCase(name);
56  		name += "_" + workField.getName();
57  		int index = -1;
58  		Field[] fields = workClass.getDeclaredFields();
59  		for (int i = 0; i < fields.length; i++) {
60  			if (field.equals(fields[i])) {
61  				index = i;
62  			}
63  		}
64  		name += "_" + index;
65  		getVariableMap().put(name, workClass);
66  		return getProvider().newVariable(name, index);
67  	}
68  
69  	public final Field toField(PrologVariable variable) {
70  		return toField(variable.getName());
71  	}
72  
73  	public final Field toField(String name) {
74  		String workName = Assertions.requireNotNull(name);
75  		Class<?> workClass = getVariableMap().get(workName);
76  		String message = name + "don't belong to any class field";
77  		workClass = Assertions.notNull(workClass, message);
78  		String className = workClass.getName();
79  		className = className.replace('.', '_') + "_";
80  		String target = JavaStrings.toUpperCase(className);
81  		workName = workName.replace(target, "");
82  		int endIndex = workName.lastIndexOf('_');
83  		workName = workName.substring(0, endIndex);
84  		return JavaReflect.getDeclaredField(workClass, workName);
85  	}
