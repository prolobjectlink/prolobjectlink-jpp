100 	public final <O> boolean contains(Predicate<O> predicate) {
101 		Class<O> clazz = classOf(predicate);
102 		PrologQuery query = prologQueryOf(clazz);
103 		if (query.hasSolution()) {
104 			Map<String, PrologTerm>[] solutionsMap = query.allVariablesSolutions();
105 			for (int i = 0; i < solutionsMap.length; i++) {
106 				O solution = (O) getConverter().toObject(clazz, solutionsMap[i]);
107 				if (predicate.evaluate(solution)) {
108 					return true;
109 				}
110 			}
111 		}
112 		return false;
113 	}

136 	public final <O> O find(O o) {
137 		Map<String, PrologTerm> inspectionMap = new HashMap<String, PrologTerm>();
138 		PrologQuery query = prologQueryOf(o, inspectionMap);
139 		if (query.hasSolution()) {
140 			Map<String, PrologTerm> solutionMap = query.oneVariablesSolution();
141 			if (!solutionMap.isEmpty()) {
142 
143 				// put all resolved by inspection
144 				solutionMap.putAll(inspectionMap);
145 				return (O) getConverter().toObject(classOf(o), solutionMap);
146 			} else {
147 				// the term exist and is equivalent to the query term
148 				return o;
149 			}
150 		}
151 		return null;
152 	}

186 	public final <O> List<O> findAll(O o) {
187 		Map<String, PrologTerm> inspectionMap = new HashMap<String, PrologTerm>();
188 		PrologTerm goal = getConverter().toTerm(o, inspectionMap);
189 		PrologQuery query = getEngine().query(goal);
190 		List<O> solutionSet = new ArrayList<O>();
191 		if (query.hasSolution()) {
192 			Map<String, PrologTerm>[] solutionsMap = query.allVariablesSolutions();
193 			if (solutionsMap.length > 0) {
194 				for (int i = 0; i < solutionsMap.length; i++) {
195 					Map<String, PrologTerm> solutionMap = solutionsMap[i];
196 					solutionMap.putAll(inspectionMap);
197 					O solution = (O) getConverter().toObject(classOf(o), solutionMap);
198 					solutionSet.add(solution);
199 				}
200 			} else {
201 				// a new term is built from the goal because
202 				// the term exist but is equivalent to the query term
203 				O solution = (O) getConverter().toObject(goal);
204 				// add to the solution as unique solution
205 				solutionSet.add(solution);
206 			}
207 		}
208 		return solutionSet;
209 	}

238 		for (PrologClause clause : getEngine()) {

265 	public final <O> void update(O match, O merge) {
266 		checkStorableObject(match);
267 		checkStorableObject(merge);
268 		checkReplacementObject(match, merge);
269 		PrologTerm pMatch = getConverter().toTerm(match);
270 		PrologTerm pMerge = getConverter().toTerm(merge);
271 		getEngine().retract(pMatch);
272 		getEngine().assertz(pMerge);
273 		setDirty(true);
274 	}
