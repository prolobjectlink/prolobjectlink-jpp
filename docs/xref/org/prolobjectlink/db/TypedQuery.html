40  /**
41   * <p>
42   * Query is a concept for retrieve objects from object containers. The query
43   * object is created through {@link Storage},{@link PersistentContainer} which
44   * have three variants {@link Storage#createQuery(Class)},
45   * {@link Storage#createQuery(Object)} (query by example ) and
46   * {@link Storage#createQuery(Predicate)}. This methods force to the prolog
47   * engine to find all possibles solution for the formulated query and with this
48   * solutions the query is builded. All before mentioned methods return a
49   * {@link TypedQuery} object that hold a solution list that can be retrieved
50   * whit the {@link TypedQuery#getSolutions()} method or simple retrieve the
51   * first solution with {@link TypedQuery#getSolution()}.
52   * </p>
53   * 
54   * <pre>
55   * TypedQuery&lt;Point&gt; query = objectContainer.createQuery(Point.class);
56   * TypedQuery&lt;Point&gt; query = objectContainer.createQuery(new Point(3, 14));
57   * </pre>
58   * 
59   * <p>
60   * {@link TypedQuery} object have some useful methods for fix the max solution
61   * numbers to retrieve and allow fix the first solution position in the internal
62   * list.
63   * </p>
64   * 
65   * <pre>
66   * query.setMaxSolution(2).setFirstSolution(2);
67   * 
68   * // fix 2 solutions number beginning at second position
69   * </pre>
70   * 
71   * <p>
72   * The internal solution list can be ordering in ascendant, descendant or by
73   * some given comparator interface implementation. See
74   * {@link TypedQuery#orderBy(Comparator)}
75   * </p>
76   * 
77   * <pre>
78   * List&lt;Point&gt; points = objectContainer.createQuery(Point.class).orderAscending().getSolutions();
79   * List&lt;Point&gt; points = objectContainer.createQuery(Point.class).orderDescending().getSolutions();
80   * </pre>
81   * 
82   * <p>
83   * {@link TypedQuery} object allow select the objects to retrieve from all
84   * solution internal list given a {@link Predicate} implementation of the match
85   * method. See TypedQuery#having(Predicate)
86   * </p>
87   * 
88   * <p>
89   * {@link TypedQuery} object allow walk over the graph formed by the class
90   * relationship described by your fields and retrieve the solutions builded with
91   * objects values from visited field. See {@link TypedQuery#descend(String)}
92   * </p>
93   * 
94   * <pre>
95   * objectContainer.createQuery(Polygon.class).descend(&quot;segment0&quot;).descend(&quot;point0&quot;).descend(&quot;idp&quot;).getSolutions();
96   * </pre>
97   * 
98   * <p>
99   * {@link TypedQuery} object implement {@link Iterator} interface providing an
100  * easy way to retrieve each object present in the internal solution list.
101  * </p>
102  * 
103  * <pre>
104  * TypedQuery&lt;Point&gt; query = container.createQuery(Point.class);
105  * for (Iterator&lt;Point&gt; i = query; e.hasNext();) {
106  * 	Point point = (Point) e.next();
107  * 	// do something with the retrieved point object
108  * }
109  * </pre>
110  * 
111  * @param <O> Root object template class
112  * 
113  * @see Query
114  * @author Jose Zalacain
115  * @since 1.0
116  * 
117  */
118 public interface TypedQuery<O> extends Iterator<O>, Serializable {

182 	/**
183 	 * <p>
184 	 * Ascendant ordering of all solutions to be returned by the solutions methods
185 	 * invocation. The order criterion is base in object hash code. Is suggested and
186 	 * very important the {@code Object#hashCode()} implementation in the model
187 	 * objects to achieve a corrected solution . The alternation with the
188 	 * {@code TypedQuery#orderDescending()} invocation only apply the latest
189 	 * ordering method invocation. For example if we want the list of stored point
190 	 * in ascendant order:
191 	 * </p>
192 	 * 
193 	 * <pre>
194 	 * List&lt;Point&gt; points = objectContainer.createQuery(Point.class).orderAscending().getSolutions();
195 	 * </pre>
196 	 * 
197 	 * <p>
198 	 * This example return a list with a, b, c, d Point class instances objects
199 	 * </p>
200 	 * 
201 	 * 
202 	 * 
203 	 * @return current instance of {@code TypedQuery} with the restriction appended.
204 	 * @since 1.0
205 	 * @see TypedQuery#orderDescending()
206 	 * @see TypedQuery#orderBy(Comparator)
207 	 */
208 	public TypedQuery<O> orderAscending();
209 
210 	/**
211 	 * Descendant ordering of all solutions to be returned by the solutions methods
212 	 * invocation. The order criterion is base in object hash code. Is suggested and
213 	 * very important the {@code Object#hashCode()} implementation in the model
214 	 * objects to achieve a corrected solution . The alternation with the
215 	 * {@code TypedQuery#orderAscending()} invocation only apply the latest ordering
216 	 * method invocation. For example if we want the list of stored point in
217 	 * descendant order:
218 	 * 
219 	 * <pre>
220 	 * List&lt;Point&gt; points = objectContainer.createQuery(Point.class).orderDescending().getSolutions();
221 	 * </pre>
222 	 * 
223 	 * <p>
224 	 * This example return a list with d, c, b, a Point class instances objects
225 	 * </p>
226 	 * 
227 	 * 
228 	 * @return current instance of {@code TypedQuery} with the restriction appended.
229 	 * @since 1.0
230 	 * @see TypedQuery#orderAscending()
231 	 * @see TypedQuery#orderBy(Comparator)
232 	 */
233 	public TypedQuery<O> orderDescending();
234 
235 	/**
236 	 * <p>
237 	 * Ordering of all solutions to be returned by the solutions methods invocation
238 	 * by a given comparator implementation. This alternative is useful when most
239 	 * specific compare criteria is needed between objects.
240 	 * </p>
241 	 * 
242 	 * <p>
243 	 * The followed example is equivalent to apply ascending order to point
244 	 * solutions based in point identifier. This example return a list with a, b, c,
245 	 * d Point class instances objects
246 	 * </p>
247 	 * 
248 	 * <pre>
249 	 * List&lt;Point&gt; points = objectContainer.createQuery(Point.class).orderBy(new Comparator&lt;Point&gt;() {
250 	 * 
251 	 * 	public int compare(Point o1, Point o2) {
252 	 * 		if (o1.getIdp().compareTo(o2.getIdp()) &lt; 0) {
253 	 * 			return -1;
254 	 * 		} else if (o1.getIdp().compareTo(o2.getIdp()) &gt; 0) {
255 	 * 			return 1;
256 	 * 		}
257 	 * 		return 0;
258 	 * 	}
259 	 * }).getSolutions();
260 	 * </pre>
261 	 * 
262 	 * @param comparator comparator used for compare an ordering solutions to be
263 	 *                   returned
264 	 * @return current instance of {@code TypedQuery} with the restriction appended.
265 	 * @since 1.0
266 	 * @see TypedQuery#orderAscending()
267 	 * @see TypedQuery#orderDescending()
268 	 */
269 	public TypedQuery<O> orderBy(Comparator<O> comparator);

279 	/**
280 	 * <p>
281 	 * Create a new {@code TypedQuery} descending to object store in the specified
282 	 * field that match with name parameter and present in the objects hold by the
283 	 * query instance. If the specified field is not found in the objects hold by
284 	 * the query instance then an exception is raised. This operation walk the graph
285 	 * represented by object relationship determined by yours field.
286 	 * </p>
287 	 * 
288 	 * <p>
289 	 * For example suppose a Polygon class is composed by segments objects and one
290 	 * segment by points objects and the point by one point identifier. If we want
291 	 * all identifiers of the initials points of the initials segments from all
292 	 * polygons:
293 	 * </p>
294 	 * 
295 	 * <pre>
296 	 * objectContainer.createQuery(Polygon.class).descend(&quot;segment0&quot;).descend(&quot;point0&quot;).descend(&quot;idp&quot;).getSolutions();
297 	 * </pre>
298 	 * 
299 	 * @param name class field name to descend
300 	 * @return new instance of {@code TypedQuery} with the restriction appended.
301 	 * @since 1.0
302 	 */
303 	public TypedQuery<Object> descend(String name);
