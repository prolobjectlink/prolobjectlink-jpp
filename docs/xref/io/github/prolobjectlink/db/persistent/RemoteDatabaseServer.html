86  	protected RemoteDatabaseServer(Settings settings, DatabaseUser systemAdmin, int serverPort) {
87  		this.threads = new HashMap<String, RemoteDatabaseThread>();
88  		this.running = Executors.newCachedThreadPool();
89  		this.systemAdmin = systemAdmin;
90  		this.serverPort = serverPort;
91  		this.settings = settings;
92  		if (serverSocket == null) {
93  			try {
94  				serverSocket = new ServerSocket(serverPort);
95  				Runtime.getRuntime().addShutdownHook(new Thread() {
96  					@Override
97  					public void run() {
98  						shutdown();
99  					}
100 				});
101 			} catch (IOException e) {
102 				LoggerUtils.error(getClass(), LoggerConstants.IO, e);
103 			}
104 		}
105 	}

107 	public Map<String, DatabaseEngine> relationals() throws IOException {
108 		Map<String, DatabaseEngine> relationals = JavaMaps.hashMap();
109 
110 		// load relational databases
111 		String relational = "db/pdb/relational";
112 		File rdat = new File(relational);
113 		File[] files = rdat.listFiles();
114 		if (files != null) {
115 			for (File file : files) {
116 				String name = file.getName();
117 
118 				// create schema instance at "name" location an load
119 				Schema schema = new DatabaseSchema(relational + "/" + name, settings.getProvider(), settings,
120 						systemAdmin).load();
121 
122 				StorageGraph storage = settings.createStorageGraph(relational + "/" + name + "/database", schema,
123 						StorageMode.STORAGE_POOL);
124 
125 				// create remote URL to embedded URL
126 				URL embeddedURL = new URL("file:/" + relational + "/" + name);
127 
128 				relationals.put(name, new EmbeddedRelational(name, embeddedURL, schema, systemAdmin, storage));
129 			}
130 		}
131 		return relationals;
132 	}

134 	public Map<String, DatabaseEngine> hierarchicals() throws IOException {
135 		Map<String, DatabaseEngine> hierarchicals = JavaMaps.hashMap();
136 
137 		// load hierarchical databases
138 		String hierarchical = "db/pdb/hierarchical";
139 		File hdat = new File(hierarchical);
140 		File[] files = hdat.listFiles();
141 		if (files != null) {
142 			for (File file : files) {
143 				String name = file.getName();
144 
145 				// create schema instance at "name" location an load
146 				Schema schema = new DatabaseSchema(hierarchical + "/" + name, settings.getProvider(), settings,
147 						systemAdmin).load();
148 
149 				// create storage instance at "name" location
150 				StorageManager storage = settings.createStorageManager(hierarchical + "/" + name + "/database",
151 						StorageMode.STORAGE_POOL);
152 
153 				// create remote URL to embedded URL
154 				URL embeddedURL = new URL("file:/" + hierarchical + "/" + name);
155 
156 				// create database using storage and schema
157 				hierarchicals.put(name, new EmbeddedHierarchical(name, embeddedURL, schema, systemAdmin, storage));
158 			}
159 		}
160 		return hierarchicals;
161 	}

188 	public final synchronized void startup() throws IOException {
189 
190 		LoggerUtils.info(getClass(), getName() + " is starting...");
191 
192 		// protocol system register
193 		System.setProperty("java.protocol.handler.pkgs", PROTOCOL);
194 
195 		// load existing databases
196 		Map<String, DatabaseEngine> hierarchicals = hierarchicals();
197 		Map<String, DatabaseEngine> relationals = relationals();
198 
199 		LoggerUtils.info(getClass(), getName() + " is started");
200 
201 		// listening loop
202 		while (!stop && serverSocket != null) {
203 			Socket socket = serverSocket.accept();
204 			String clientAddress = socket.getInetAddress().getHostAddress();
205 			RemoteDatabaseThread thread = threads.get(clientAddress);
206 			if (thread == null) {
207 				thread = new RemoteDatabaseThread(threadId++, socket, this);
208 				thread.putDatabases(DatabaseType.HIERARCHICAL, hierarchicals);
209 				thread.putDatabases(DatabaseType.RELATIONAL, relationals);
210 				threads.put(clientAddress, thread);
211 			}
212 			thread.setSocket(socket);
213 			thread.setServer(this);
214 			running.submit(thread);
215 		}
216 
217 		LoggerUtils.info(getClass(), getName() + " is shutting down...");
218 
219 		// close all databases after stop listening loop
220 		for (DatabaseEngine e : hierarchicals.values()) {
221 			e.close();
222 		}
223 		hierarchicals.clear();
224 		for (DatabaseEngine e : relationals.values()) {
225 			e.close();
226 		}
227 		relationals.clear();
228 
229 		// shutdown the executor
230 		running.shutdown();
231 
232 		// clear active threads
233 		threads.clear();
234 
235 		// close server socket
236 		if (serverSocket != null) {
237 			serverSocket.close();
238 			serverSocket = null;
239 		}
240 
241 		LoggerUtils.info(getClass(), getName() + " is shutdown");
242 
243 	}
